#instructions for setting up folders and conda enviornment

##create a new cosmosis conda environment and folders
git clone https://github.com/icanswim/cosmosis.git
cd cosmosis
mkdir data
mkdir logs
mkdir models
conda env create -f environment.yml
conda activate cosmo
jupyter lab

##instructions for the manual creation of conda environment
git clone https://github.com/icanswim/cosmosis.git
cd cosmosis
mkdir data
mkdir logs
mkdir models
conda create --name cosmo python pytorch torchvision torchaudio pytorch-cuda=11.8 -c pytorch -c nvidia
conda activate cosmo
conda install scipy pandas matplotlib scikit-learn -c conda-forge
conda install jupyterlab h5py git -c conda-forge
jupyter lab
