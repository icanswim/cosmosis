#instructions for setting up folders and conda enviornment

##create a new cosmosis conda environment and folders
git clone https://github.com/icanswim/cosmosis.git
cd cosmosis
mkdir data
mkdir logs
mkdir models
conda env create -f environment.yml
conda activate cosmo
jupyter lab

##instructions for the manual creation of conda enviornment
git clone https://github.com/icanswim/cosmosis.git
cd cosmosis
mkdir data
mkdir logs
mkdir models
conda create --name cosmo numpy pandas scipy python=3.9 -c conda-forge
conda activate cosmo
conda install pytorch torchvision torchaudio pytorch-cuda=11.8 -c pytorch -c nvidia
conda install jupyterlab -c conda-forge
conda install matplotlib -c conda-forge
conda install scikit-learn -c conda-forge
conda install h5py -c conda-forge
conda install git -c conda-forge
jupyter lab
