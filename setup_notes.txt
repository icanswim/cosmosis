#instructions for setting up folders and conda enviornment

##create a new cosmosis conda environment and folders
git clone https://github.com/icanswim/cosmosis.git
cd cosmosis
mkdir data
mkdir logs
mkdir models
conda env create -f environment.yml
conda activate cosmosis
jupyter lab

##instructions for the manual creation of conda enviornment
git clone https://github.com/icanswim/cosmosis.git
cd cosmosis
mkdir data
mkdir logs
mkdir models
conda create --name cosmo -c conda-forge python==3.8 numpy pandas scipy
conda activate cosmo
conda install pytorch==1.8.0 torchvision==0.9.0 torchaudio==0.8.0 cudatoolkit==10.2 -c pytorch
conda install -c conda-forge jupyterlab
conda install -c conda-forge matplotlib
conda install -c conda-forge scikit-learn
conda install -c conda-forge h5py
conda install -c conda-forge git
jupyter lab
